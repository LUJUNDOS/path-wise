## 四、前端 API 调用封装

### 4.1 API 客户端类型安全封装

```typescript
// apps/web/src/lib/api-client.ts

import axios, { AxiosResponse } from 'axios';
import type {
  TripConfig,
  Trip,
  TransportSearchRequest,
  TransportOptionDetail,
} from '@shared/types';

// 创建 axios 实例
const apiClient = axios.create({
  baseURL: import.meta.env.VITE_API_URL || 'http://localhost:3000',
  timeout: 30000,
  headers: {
    'Content-Type': 'application/json',
  },
});

// 类型安全的 API 调用
export const api = {
  // 交通查询
  async searchTransport(request: TransportSearchRequest): Promise<TransportOptionDetail[]> {
    const response: AxiosResponse<{
      code: number;
      data: { options: TransportOptionDetail[] };
    }> = await apiClient.post('/api/v1/transport/search', request);

    if (response.data.code !== 0) {
      throw new Error(`API Error: ${response.data.code}`);
    }

    return response.data.data.options;
  },

  // 生成攻略（SSE 流式）
  async *generateTrip(config: TripConfig): AsyncGenerator<SSEEvent> {
    const response = await fetch(`${apiClient.defaults.baseURL}/api/v1/trip/generate`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({ config }),
    });

    if (!response.ok) {
      throw new Error(`HTTP Error: ${response.status}`);
    }

    const reader = response.body!.getReader();
    const decoder = new TextDecoder();

    while (true) {
      const { done, value } = await reader.read();
      if (done) break;

      const chunk = decoder.decode(value, { stream: true });
      const events = parseSSEChunk(chunk);

      for (const event of events) {
        yield event;
      }
    }
  },

  // 获取行程详情
  async getTrip(tripId: number): Promise<Trip> {
    const response = await apiClient.get<{ code: number; data: Trip }>(`/api/v1/trip/${tripId}`);

    if (response.data.code !== 0) {
      throw new Error(`API Error: ${response.data.code}`);
    }

    return response.data.data;
  },
};

// SSE 数据块解析
function parseSSEChunk(chunk: string): SSEEvent[] {
  const events: SSEEvent[] = [];
  const lines = chunk.split('\n');

  let currentEvent = '';
  let currentData = '';

  for (const line of lines) {
    if (line.startsWith('event:')) {
      currentEvent = line.slice(6).trim();
    } else if (line.startsWith('data:')) {
      currentData = line.slice(5).trim();

      if (currentEvent && currentData) {
        events.push({
          type: currentEvent,
          data: JSON.parse(currentData),
        });

        currentEvent = '';
        currentData = '';
      }
    }
  }

  return events;
}
```

### 5.2 后端 SSE 发送实现

```typescript
// apps/api/src/services/sse.service.ts

import { ServerResponse } from 'http';
import { SSEEventType, SSEProgressData, SSEDayPlanData } from '../types/sse';

export class SSEService {
  private response: ServerResponse;

  constructor(response: ServerResponse) {
    this.response = response;

    // 设置 SSE 响应头
    response.writeHead(200, {
      'Content-Type': 'text/event-stream',
      'Cache-Control': 'no-cache',
      Connection: 'keep-alive',
    });
  }

  // 发送进度事件
  sendProgress(data: SSEProgressData): void {
    this.sendEvent(SSEEventType.PROGRESS, data);
  }

  // 发送每日行程事件
  sendDayPlan(data: SSEDayPlanData): void {
    this.sendEvent(SSEEventType.DAY_PLAN, data);
  }

  // 发送完成事件
  sendComplete(data: SSECompleteData): void {
    this.sendEvent(SSEEventType.COMPLETE, data);
    this.response.end();
  }

  // 发送错误事件
  sendError(code: number, message: string, details?: string): void {
    this.sendEvent(SSEEventType.ERROR, { code, message, details });
    this.response.end();
  }

  // 发送 SSE 事件（私有方法）
  private sendEvent(type: string, data: any): void {
    const event = `event: ${type}\n`;
    const dataStr = `data: ${JSON.stringify(data)}\n\n`;

    this.response.write(event + dataStr);
  }
}
```

### 6.2 错误处理中间件（Fastify）

```typescript
// apps/api/src/plugins/error-handler.ts

import { FastifyError, FastifyReply, FastifyRequest } from 'fastify';
import { ErrorCode, ErrorResponse } from '@shared/types';

export const errorHandler = (error: FastifyError, request: FastifyRequest, reply: FastifyReply) => {
  const errorResponse: ErrorResponse = {
    code: (error.statusCode as ErrorCode) || ErrorCode.INTERNAL_SERVER_ERROR,
    message: error.message,
    details: error.stack,
    timestamp: new Date().toISOString(),
    path: request.url,
  };

  // 记录错误日志
  request.log.error(error);

  // 发送错误响应
  reply.status(error.statusCode || 500).send(errorResponse);
};
```

## 九、类型安全验证

### 9.1 前后端类型同步策略

**问题**：如何保证前后端类型定义一致？

**解决方案**：

1. **共享类型包**（`packages/shared`）：
   - 前后端都引用 `@shared/types`
   - 单一来源（Single Source of Truth）

2. **自动化类型生成**：
   - 后端使用 Prisma，自动生成数据库模型类型
   - 使用 `prisma generate` 生成 `@prisma/client`

3. **API 契约测试**：
   - 使用 Jest 编写契约测试
   - 验证后端响应符合前端期望的类型

### 9.2 类型检查脚本
