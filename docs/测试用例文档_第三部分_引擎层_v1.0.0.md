```typescript
// packages/backend/tests/engine-init.test.ts

describe("算法一：时间轴初始化", () => {
  it("单城市应生成正确数量的天计划", () => {
    const result = initializeTimeline({
      departureCity: "beijing",
      destinations: [{ cityId: "beijing", stayDays: 3 }],
    });
    expect(result.length).toBe(3);
    expect(result[0].dayType).toBe("explore");
  });

  it("多城市应在正确位置插入 transit 日", () => {
    const result = initializeTimeline({
      departureCity: "beijing",
      destinations: [
        { cityId: "beijing", stayDays: 2 },
        { cityId: "shanghai", stayDays: 2 },
      ],
    });
    // Day 0: explore(北京), Day 1: explore(北京), Day 2: transit(北京→上海), Day 3: explore(上海)
    expect(result[2].dayType).toBe("transit");
    expect(result[2].cityId).toBe("shanghai"); // 中转日归属下一城市
  });

  it("出发城市≠首目的地时应插入 move 日", () => {
    const result = initializeTimeline({
      departureCity: "guangzhou",
      destinations: [{ cityId: "beijing", stayDays: 3 }],
    });
    expect(result[0].dayType).toBe("move");
  });
});
```

```typescript
// packages/backend/tests/engine-fill.test.ts

describe("算法三：时间轴填充", () => {
  it("必游景点不应被遗漏", () => {
    const dayPlan = { dayIndex: 0, dayType: "explore", cityId: "beijing" };
    const pool = buildCandidatePool("beijing");
    const filled = fillTimeline(dayPlan, pool);
    const mustSeePoiIds = pool.filter((p) => p.isMustSee).map((p) => p.id);
    const filledPoiIds = filled.timelineItems
      .filter((t) => t.type === "poi")
      .map((t) => t.poiId);
    mustSeePoiIds.forEach((id) => expect(filledPoiIds).toContain(id));
  });

  it("不应安排超过可用时间", () => {
    const dayPlan = {
      dayIndex: 0,
      dayType: "explore",
      cityId: "beijing",
      availableHours: 8,
    };
    const pool = buildCandidatePool("beijing");
    const filled = fillTimeline(dayPlan, pool);
    const totalTime = filled.timelineItems.reduce(
      (sum, item) => sum + item.duration,
      0,
    );
    expect(totalTime).toBeLessThanOrEqual(8 * 60); // 分钟
  });
});
```
