// e2e/single-city.spec.ts

describe('端到端：单城市流程', () => {
  it('应完成从首页到结果页的完整流程', () => {
    // 1. 打开首页
    cy.visit('/');
    // 2. 填写表单
    cy.get('[data-testid="departure-city"]').type('广州');
    cy.get('[data-testid="add-destination"]').click();
    cy.get('[data-testid="destination-city-0"]').type('北京');
    cy.get('[data-testid="stay-days-0"]').type('3');
    // 3. 点击生成
    cy.get('[data-testid="generate-btn"]').click();
    // 4. 等待 SSE 完成（最多 5 分钟）
    cy.waitUntil(() => cy.get('[data-testid="all-complete"]').should('be.visible'), {
      timeout: 300000,
    });
    // 5. 验证结果页
    cy.get('[data-testid="day-card"]').should('have.length', 3);
    cy.get('[data-testid="total-cost"]').should('not.have.text', '0');
  });
});

describe('端到端：SSE 断点续传', () => {

    cy.visit('/');

describe('性能测试', () => {
  it('单天生成耗时应 < 60 秒', () => {
    const start = Date.now();
    generateTrip({ departure: '广州', destination: '北京', days: 1 });
    cy.waitUntil(() => cy.get('[data-testid="day-card"]').should('be.visible'), { timeout: 60000 });
    const elapsed = Date.now() - start;
    expect(elapsed).to.be.lessThan(60000);
  });

  it('API 平均响应时间应 < 200ms', () => {
    const requests: number[] = [];
    cy.intercept('GET', '/api/trips', (req) => {
      const start = Date.now();
      req.continue((res) => {
        requests.push(Date.now() - start);
      });
    });
    cy.visit('/my-trips');
    cy.wait('@getTrips');
    const avg = requests.reduce((a, b) => a + b, 0) / requests.length;
    expect(avg).to.be.lessThan(200);
  });
});

### 6.9 E2E-009 安全测试

#### 测试用例

| 用例ID     | 用例描述                      | 操作步骤                                  | 预期结果                                   |
| ---------- | ----------------------------- | ----------------------------------------- | ------------------------------------------ |
| E2E-009-01 | 无 token 访问受保护接口       | 不带 Authorization Header 调用 GET /trips | 返回 401                                   |
| E2E-009-02 | 篡改 token 访问他人行程       | 使用 User-A 的 token 访问 User-B 的行程   | 返回 403                                   |
| E2E-009-03 | XSS 攻击注入                  | 在昵称中输入 `<script>alert(1)</script>`  | 存储后前端渲染时不被执行（React 默认转义） |
| E2E-009-04 | SQL 注入                      | 在目的地城市输入 `' OR 1=1--`             | 后端使用 Prisma（参数化查询），不被注入    |
| E2E-009-05 | 限流生效                      | 同一 IP 发送 101 次请求/分钟              | 第 101 次返回 429                          |
| E2E-009-06 | JWT 过期后无法访问            | 使用过期 token 访问受保护接口             | 返回 401，前端跳转登录页                   |
| E2E-009-07 | 分享链接 token 泄露后可被访问 | 知道 shareToken 即可访问                  | 返回 200（设计如此，分享链接无额外认证）   |
| E2E-009-08 | 软删除后数据不泄露            | 用户 A 软删除行程，用户 B 尝试访问        | 返回 404（如同不存在）                     |

#### 安全测试脚本

```typescript
// e2e/security.spec.ts

describe('安全测试', () => {
  it('无 token 时应返回 401', () => {
    cy.request({ url: '/api/trips', failOnStatusCode: false }).then((res) => {
      expect(res.status).to.eq(401);
    });
  });

  it('应拒绝 XSS 攻击', () => {
    // 尝试注入 XSS
    cy.request({
      method: 'PUT',
      url: '/api/user/profile',
      headers: { Authorization: `Bearer ${token}` },
      body: { nickname: '<script>alert(1)</script>' },
      failOnStatusCode: false,
    }).then((res) => {
      // 存储成功（服务端存储原始值），但前端渲染时应转义
      // 检查前端是否转义
      cy.visit('/my-profile');
      cy.get('[data-testid="nickname"]').should('not.contain.html', '<script>');
    });
  });

  it('应拒绝 SQL 注入', () => {
    cy.request({
      method: 'POST',
      url: '/api/trips/validate',
      body: { departureCity: "' OR 1=1--", destinations: [] },
      failOnStatusCode: false,
    }).then((res) => {
      expect(res.status).to.eq(400); // 校验失败，而非 SQL 注入成功
    });
  });
});
```
