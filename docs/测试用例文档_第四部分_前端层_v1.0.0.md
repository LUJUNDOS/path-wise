```typescript
// apps/web/tests/homepage.test.tsx

describe('首页整体布局', () => {
  it('桌面端应渲染居中卡片', () => {
    global.innerWidth = 1200;
    render(<HomePage />);
    const card = screen.getByTestId('home-card');
    expect(card).toHaveStyle({ 'max-width': '720px', margin: '0 auto' });
  });

  it('移动端生成按钮应 sticky 底部', () => {
    global.innerWidth = 375;
    render(<HomePage />);
    const btn = screen.getByTestId('generate-btn');

  });

  it('总天数应实时更新', () => {
    render(<HomePage />);
    fireEvent.change(screen.getByTestId('stay-days-0'), { target: { value: '3' } });
    expect(screen.getByTestId('total-days')).toHaveTextContent('共 3 天');
  });
});
```

```typescript
// apps/web/tests/sse-manager.test.tsx

describe('SSE 连接管理', () => {
  it('应正确流转状态：idle→connecting→streaming→all_complete', () => {
    const { result } = renderHook(() => useSSE());
    expect(result.current.status).toBe('idle');
    act(() => result.current.connect());
    expect(result.current.status).toBe('connecting');
    // Mock SSE 事件
    act(() => result.current.mockEvent('connected', {}));
    expect(result.current.status).toBe('streaming');
  });
});
```
