## 三、前端错误处理

### 3.1 组件级错误处理

#### 使用 Error Boundary

```tsx
// components/common/ErrorBoundary.tsx

import { Component, ErrorInfo, ReactNode } from "react";
import { Button } from "@/components/ui/button";
import { AlertTriangle } from "lucide-react";

interface ErrorBoundaryProps {
  children: ReactNode;
  fallback?: (error: Error, reset: () => void) => ReactNode;
}

interface ErrorBoundaryState {
  hasError: boolean;
  error?: Error;
}

export class ErrorBoundary extends Component<
  ErrorBoundaryProps,
  ErrorBoundaryState
> {
  constructor(props: ErrorBoundaryProps) {
    super(props);
    this.state = { hasError: false };
  }

  static getDerivedStateFromError(error: Error): ErrorBoundaryState {
    return { hasError: true, error };
  }

  componentDidCatch(error: Error, errorInfo: ErrorInfo): void {
    // 记录错误日志
    console.error("组件错误:", error, errorInfo);

    // 发送错误到监控服务（如 Sentry）
    // sendErrorToMonitoring(error, errorInfo);
  }

  render(): ReactNode {
    if (this.state.hasError) {
      // 使用自定义 fallback UI
      if (this.props.fallback) {
        return this.props.fallback(this.state.error!, () => {
          this.setState({ hasError: false, error: undefined });
        });
      }

      // 默认 fallback UI
      return (
        <div className="flex flex-col items-center justify-center min-h-[400px] p-8">
          <AlertTriangle className="h-16 w-16 text-destructive mb-4" />
          <h2 className="text-2xl font-bold mb-2">页面出错了</h2>
          <p className="text-muted-foreground mb-4">
            {this.state.error?.message || "未知错误"}
          </p>
          <Button
            onClick={() => {
              this.setState({ hasError: false, error: undefined });
            }}
          >
            重试
          </Button>
        </div>
      );
    }

    return this.props.children;
  }
}
```


#### API 错误拦截器

```typescript
// apps/web/src/lib/api-client.ts

import axios, { AxiosError } from "axios";
import { ErrorCode, ErrorMessageMap } from "@shared/types";

// 创建 axios 实例
const apiClient = axios.create({
  baseURL: import.meta.env.VITE_API_URL,
  timeout: 30000,
});

// 响应拦截器（全局错误处理）
apiClient.interceptors.response.use(
  (response) => response,
  (error: AxiosError<ErrorResponse>) => {
    // 网络错误
    if (!error.response) {
      console.error("网络错误:", error);
      showToast("网络连接失败，请检查网络设置", "error");
      return Promise.reject(error);
    }

    // API 错误
    const errorResponse = error.response.data;
    const errorCode = errorResponse?.code;
    const errorMessage = ErrorMessageMap[errorCode as ErrorCode] || "未知错误";

    // 记录错误日志
    console.error("API 错误:", {
      code: errorCode,
      message: errorMessage,
      details: errorResponse?.details,
      url: error.config?.url,
    });

    // 显示用户友好的错误提示
    showToast(errorMessage, "error");

    // 特殊错误处理
    if (errorCode === ErrorCode.UNAUTHORIZED) {
      // 跳转到登录页
      window.location.href = "/login";
    }

    return Promise.reject(error);
  },
);
```

#### Toast 通知组件

```tsx
// components/common/Toast.tsx

import { useState, useEffect } from "react";
import { X } from "lucide-react";

type ToastType = "success" | "error" | "warning" | "info";

interface Toast {
  id: number;
  message: string;
  type: ToastType;
  duration?: number; // 自动关闭时间（毫秒）
}

let toastId = 0;
const toasts: Toast[] = [];
const listeners: (() => void)[] = [];

// 显示 Toast
export function showToast(
  message: string,
  type: ToastType = "info",
  duration = 3000,
): void {
  const toast: Toast = {
    id: ++toastId,
    message,
    type,
    duration,
  };

  toasts.push(toast);
  listeners.forEach((listener) => listener());

  // 自动关闭
  if (duration > 0) {
    setTimeout(() => {
      removeToast(toast.id);
    }, duration);
  }
}

// 移除 Toast
export function removeToast(id: number): void {
  const index = toasts.findIndex((t) => t.id === id);
  if (index > -1) {
    toasts.splice(index, 1);
    listeners.forEach((listener) => listener());
  }
}

// Toast 容器组件
export function ToastContainer() {
  const [toastList, setToastList] = useState<Toast[]>([]);

  useEffect(() => {
    const listener = () => setToastList([...toasts]);
    listeners.push(listener);
    return () => {
      const index = listeners.indexOf(listener);
      if (index > -1) listeners.splice(index, 1);
    };
  }, []);

  return (
    <div className="fixed top-4 right-4 z-50 flex flex-col gap-2">
      {toastList.map((toast) => (
        <div
          key={toast.id}
          className={cn(
            "flex items-center justify-between p-4 rounded-lg shadow-lg min-w-[300px]",
            toast.type === "success" && "bg-green-500 text-white",
            toast.type === "error" && "bg-red-500 text-white",
            toast.type === "warning" && "bg-yellow-500 text-black",
            toast.type === "info" && "bg-blue-500 text-white",
          )}
        >
          <span>{toast.message}</span>
          <button onClick={() => removeToast(toast.id)}>
            <X className="h-4 w-4" />
          </button>
        </div>
      ))}
    </div>
  );
}
```

### 4.1 Fastify 错误处理器

```typescript
// apps/api/src/plugins/error-handler.ts

import { FastifyError, FastifyReply, FastifyRequest } from "fastify";
import { ErrorCode, ErrorResponse } from "@shared/types";

// 全局错误处理器
export const errorHandler = (
  error: FastifyError,
  request: FastifyRequest,
  reply: FastifyReply,
): void => {
  // 确定错误码
  let errorCode: ErrorCode;
  let message: string;
  let details: string | undefined;

  // Fastify 内置错误
  if (error.statusCode) {
    errorCode = error.statusCode as ErrorCode;
    message = error.message;
  } else if (error.code === "FST_ERR_VALIDATION") {
    // 参数验证错误
    errorCode = ErrorCode.BAD_REQUEST;
    message = "请求参数验证失败";
    details = JSON.stringify(error.validation);
  } else {
    // 未知错误
    errorCode = ErrorCode.INTERNAL_SERVER_ERROR;
    message = "服务器内部错误";
    details = error.stack;
  }

  // 记录错误日志
  request.log.error({
    code: errorCode,
    message: error.message,
    stack: error.stack,
    url: request.url,
    method: request.method,
  });

  // 构建错误响应
  const errorResponse: ErrorResponse = {
    code: errorCode,
    message,
    details: process.env.NODE_ENV === "production" ? undefined : details,
    timestamp: new Date().toISOString(),
    path: request.url,
  };

  // 发送错误响应
  reply.status(errorCode).send(errorResponse);
};
```

### 4.3 LLM API 错误处理

```typescript
// apps/api/src/services/llm.service.ts

import { LLMAPIError } from "../types/errors";

export class LLMService {
  async generate(prompt: string, model = "deepseek-v2"): Promise<string> {
    try {
      const response = await fetch(`https://api.${model}.com/generate`, {
        method: "POST",
        headers: {
          Authorization: `Bearer ${process.env[`${model.toUpperCase()}_API_KEY`]}`,
          "Content-Type": "application/json",
        },
        body: JSON.stringify({ prompt }),
      });

      if (!response.ok) {
        const errorText = await response.text();
        throw new LLMAPIError(
          `LLM API 调用失败: ${response.status}`,
          errorText,
        );
      }

      return await response.text();
    } catch (error) {
      // 重试逻辑
      if (error instanceof LLMAPIError && this.shouldRetry(error)) {
        return this.retry(() => this.generate(prompt, model), 3);
      }

      throw error;
    }
  }

  // 判断是否可重试
  private shouldRetry(error: LLMAPIError): boolean {
    // 限流错误（429）可重试
    return (
      error.message.includes("429") || error.message.includes("rate_limit")
    );
  }

  // 重试逻辑（指数退避）
  private async retry<T>(fn: () => Promise<T>, maxRetries: number): Promise<T> {
    let lastError: Error;

    for (let i = 0; i < maxRetries; i++) {
      try {
        return await fn();
      } catch (error) {
        lastError = error as Error;

        // 等待后重试（指数退避）
        const delayMs = Math.min(1000 * Math.pow(2, i), 30000);
        await new Promise((resolve) => setTimeout(resolve, delayMs));
      }
    }

    throw lastError!;
  }
}
```

### 5.1 错误提示原则

| 原则       | 说明           | 示例                                                   |
| ---------- | -------------- | ------------------------------------------------------ |
| **明确**   | 说明什么问题   | ❌ "出错啦" <br> ✅ "出发日期不能晚于返回日期"         |

```tsx
  warning: <AlertTriangle className="h-5 w-5" />,

      {onClose && (
        <button onClick={onClose} className="flex-shrink-0">

    </div>

}
```

## 七、错误恢复策略

### 7.1 前端错误恢复

| 错误类型            | 恢复策略                   | 示例                       |
| ------------------- | -------------------------- | -------------------------- |
| **网络错误**        | 自动重试（指数退避）       | Axios 重试拦截器           |
| **API 错误（4xx）** | 提示用户修正输入           | 表单验证错误               |
| **API 错误（5xx）** | 显示错误提示，提供重试按钮 | "服务暂不可用，请稍后重试" |
| **组件错误**        | Error Boundary 降级展示    | 显示 fallback UI           |

### 7.2 后端错误恢复

| 错误类型            | 恢复策略           | 示例                   |
| ------------------- | ------------------ | ---------------------- |
| **LLM API 限流**    | 自动重试（等待后） | 指数退避重试           |
| **数据库错误**      | 事务回滚           | Prisma 事务            |
| **第三方 API 错误** | 降级到静态数据     | 使用城市知识库静态数据 |
| **SSE 连接中断**    | 前端自动重连       | SSEClient 重连逻辑     |

```typescript
// apps/web/src/components/common/ErrorBoundary.test.tsx

import { render, screen, fireEvent } from '@testing-library/react';
import { ErrorBoundary } from './ErrorBoundary';

// 故意抛出错误的组件

    throw new Error('测试错误');
  }
  return <div>正常组件</div>;
}

describe('ErrorBoundary', () => {
  it('应捕获并显示错误信息', () => {
    render(
      <ErrorBoundary>
        <ThrowError shouldThrow={true} />
      </ErrorBoundary>
    );

    expect(screen.getByText('页面出错了')).toBeInTheDocument();
  });

  it('应显示重试按钮', () => {
    render(
      <ErrorBoundary>
        <ThrowError shouldThrow={true} />
      </ErrorBoundary>
    );

    const retryButton = screen.getByText('重试');
    expect(retryButton).toBeInTheDocument();

    // 点击重试按钮
    fireEvent.click(retryButton);
    expect(screen.getByText('正常组件')).toBeInTheDocument();
  });
});
```
