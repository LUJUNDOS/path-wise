## 三、核心组件设计

### 3.1 城市选择器（CitySelector）

#### Props 接口

```typescript
// components/trip/CitySelector.tsx

interface CitySelectorProps {
  // 数据源
  cities: CityBasic[]; // 城市列表（从城市知识库加载）

  // 选中值
  value: string[]; // 选中的城市（支持多目的地）
  onChange: (cities: string[]) => void;

  // 模式
  mode: 'single' | 'multiple'; // 单目的地 / 多目的地
  maxCities?: number; // 最多选择城市数（mode='multiple' 时）

  // UI
  placeholder?: string;
  disabled?: boolean;

  // 筛选
  showTags?: boolean; // 是否显示标签筛选（美食/自然/历史...）
  popularCities?: string[]; // 热门城市（优先显示）
}
```

#### 组件状态

```typescript
interface CitySelectorState {
  isOpen: boolean; // 下拉框是否打开
  searchQuery: string; // 搜索关键词
  selectedTags: string[]; // 选中的标签（筛选）
  filteredCities: CityBasic[]; // 筛选后的城市列表
}
```

#### 交互流程

```
用户点击城市输入框
  │
  ▼
展开下拉框（isOpen = true）
  │
  ▼
显示城市列表（按热门/标签分组）
  │
  ├─ 用户输入搜索关键词
  │   └─ 实时筛选城市列表（filteredCities）
  │
  ├─ 用户点击标签（如"美食"）
  │   └─ 筛选包含该标签的城市
  │
  ├─ 用户点击城市
  │   ├─ mode='single' → 直接选中，关闭下拉框
  │   └─ mode='multiple' → 添加到 value 数组，继续选择
  │
  └─ 用户点击"确认"
      └─ 触发 onChange，关闭下拉框
```

#### 实现要点

```typescript
// components/trip/CitySelector.tsx

export function CitySelector({
  cities,
  value,
  onChange,
  mode = 'single',
  maxCities = 5,
  placeholder = '选择城市...',
  showTags = true,
  popularCities = ['北京', '上海', '成都', '杭州', '西安'],
}: CitySelectorProps) {
  const [isOpen, setIsOpen] = useState(false);
  const [searchQuery, setSearchQuery] = useState('');
  const [selectedTags, setSelectedTags] = useState<string[]>([]);

  // 筛选城市列表
  const filteredCities = useMemo(() => {
    let result = cities;

    // 按搜索关键词筛选
    if (searchQuery) {
      result = result.filter(city =>
        city.cityName.includes(searchQuery) ||
        city.province?.includes(searchQuery)
      );
    }

    // 按标签筛选
    if (selectedTags.length > 0) {
      result = result.filter(city =>
        selectedTags.some(tag => city.tags.includes(tag))
      );
    }

    // 排序：热门城市优先
    result.sort((a, b) => {
      const aIsPopular = popularCities.includes(a.cityName);
      const bIsPopular = popularCities.includes(b.cityName);

      if (aIsPopular && !bIsPopular) return -1;
      if (!aIsPopular && bIsPopular) return 1;
      return 0;
    });

    return result;
  }, [cities, searchQuery, selectedTags, popularCities]);

  // 处理城市选择
  const handleCitySelect = (cityName: string) => {
    if (mode === 'single') {
      onChange([cityName]);
      setIsOpen(false);
    } else {
      if (value.includes(cityName)) {
        // 取消选中
        onChange(value.filter(c => c !== cityName));
      } else {
        // 选中（检查上限）
        if (value.length >= maxCities) {
          alert(`最多选择 ${maxCities} 个城市`);
          return;
        }
        onChange([...value, cityName]);
      }
    }
  };

  return (
    <Popover open={isOpen} onOpenChange={setIsOpen}>
      <PopoverTrigger asChild>
        <Button variant="outline" className="w-full justify-between">
          {value.length > 0
            ? value.join(' → ')
            : placeholder}
          <ChevronDown className="ml-2 h-4 w-4" />
        </Button>
      </PopoverTrigger>
      <PopoverContent className="w-[400px] p-4">
        {/* 搜索框 */}
        <Input
          placeholder="搜索城市..."
          value={searchQuery}
          onChange={(e) => setSearchQuery(e.target.value)}
          className="mb-4"
        />

        {/* 标签筛选 */}
        {showTags && (
          <div className="mb-4 flex flex-wrap gap-2">
            {['美食', '自然', '历史', '现代'].map(tag => (
              <Badge
                key={tag}
                variant={selectedTags.includes(tag) ? 'default' : 'outline'}
                className="cursor-pointer"
                onClick={() => {
                  if (selectedTags.includes(tag)) {
                    setSelectedTags(selectedTags.filter(t => t !== tag));
                  } else {
                    setSelectedTags([...selectedTags, tag]);
                  }
                }}
              >
                {tag}
              </Badge>
            ))}
          </div>
        )}

        {/* 城市列表 */}
        <div className="max-h-[300px] overflow-y-auto">
          {filteredCities.map(city => (
            <div
              key={city.cityName}
              className={cn(
                'flex items-center justify-between p-2 rounded cursor-pointer hover:bg-accent',
                value.includes(city.cityName) && 'bg-accent'
              )}
              onClick={() => handleCitySelect(city.cityName)}
            >
              <div>
                <p className="font-medium">{city.cityName}</p>
                <p className="text-sm text-muted-foreground">{city.province}</p>
              </div>
              <div className="flex gap-1">
                {city.tags.map(tag => (
                  <Badge key={tag} variant="secondary" className="text-xs">
                    {tag}
                  </Badge>
                ))}
              </div>
            </div>
          ))}
        </div>
      </PopoverContent>
    </Popover>
  );
}
```

```typescript
// components/itinerary/DayPlanCard.tsx

export function DayPlanCard({
  dayPlan,
  isGenerating = false,
  onAccommodationClick,
  onPOIClick,
  onEdit,
}: DayPlanCardProps) {
  return (
    <Card className="mb-6">
      <CardHeader>
        <div className="flex items-center justify-between">
          <CardTitle>{dayPlan.title}</CardTitle>
          {!isGenerating && onEdit && (
            <Button variant="ghost" size="sm" onClick={() => onEdit(dayPlan.day)}>
              <Edit className="h-4 w-4" />
            </Button>
          )}
        </div>
        <CardDescription>{dayPlan.date} · {dayPlan.summary}</CardDescription>
      </CardHeader>
      <CardContent>
        {/* 大交通信息 */}
        {dayPlan.transport && (
          <TransportInfoCard transport={dayPlan.transport} className="mb-4" />
        )}

        {/* 时间段 */}
        <div className="space-y-4">
          {dayPlan.morning && (
            <TimeSlotCard
              title="上午"
              timeSlot={dayPlan.morning}
              onPOIClick={onPOIClick}
            />
          )}
          {dayPlan.afternoon && (
            <TimeSlotCard
              title="下午"
              timeSlot={dayPlan.afternoon}
              onPOIClick={onPOIClick}
            />
          )}
          {dayPlan.evening && (
            <TimeSlotCard
              title="晚上"
              timeSlot={dayPlan.evening}
              onPOIClick={onPOIClick}
            />
          )}
        </div>

        {/* 住宿信息 */}
        {dayPlan.accommodation && (
          <AccommodationCard
            accommodation={dayPlan.accommodation}
            className="mt-4"
            onClick={() => onAccommodationClick?.(dayPlan.accommodation!)}
          />
        )}

        {/* 注意事项 */}
        {dayPlan.notes && dayPlan.notes.length > 0 && (
          <Alert className="mt-4">
            <AlertCircle className="h-4 w-4" />
            <AlertDescription>
              <ul className="list-disc list-inside">
                {dayPlan.notes.map((note, index) => (
                  <li key={index}>{note}</li>
                ))}
              </ul>
            </AlertDescription>
          </Alert>
        )}
      </CardContent>
    </Card>
  );
}
```

```typescript
// components/itinerary/TransportInfoCard.tsx

const TRANSPORT_ICONS: Record<TransportType, string> = {
  'high_speed_rail': '🚄',
  'normal_train': '🚃',
  'flight': '✈️',
  'bus': '🚌',
  'auto': '🚗',
};

export function TransportInfoCard({ transport, className }: TransportInfoCardProps) {
  const icon = TRANSPORT_ICONS[transport.type];
  const isOvernight = transport.isOvernight;

  return (
    <Card className={cn('border-l-4 border-l-blue-500', className)}>
      <CardContent className="pt-6">
        <div className="flex items-start gap-4">
          {/* Icon */}
          <div className="text-4xl">{icon}</div>

          {/* 信息 */}
          <div className="flex-1">
            <div className="flex items-center gap-2">
              <h4 className="font-semibold">
                {transport.trainNumber || transport.flightNumber}
              </h4>
              {isOvernight && (
                <Badge variant="secondary">💤 隔夜</Badge>
              )}
            </div>

            <p className="text-sm text-muted-foreground mt-1">
              {transport.departureStation} {transport.departTime} →{' '}
              {transport.arrivalStation} {transport.arrivalTime}
            </p>

            <p className="text-sm mt-2">
              历时 {Math.floor(transport.duration / 60)} 小时 {transport.duration % 60} 分钟
            </p>

            {/* 价格 */}
            <div className="mt-2 flex gap-4">
              {Object.entries(transport.price).map(([seatType, price]) => (
                <div key={seatType} className="text-sm">
                  <span className="text-muted-foreground">{seatType}：</span>
                  <span className="font-medium">¥{price}</span>
                </div>
              ))}
            </div>
          </div>
        </div>

        {/* 免责声明 */}
        <Alert className="mt-4 border-orange-500">
          <AlertCircle className="h-4 w-4 text-orange-500" />
          <AlertDescription className="text-orange-500 text-sm">
```

## 四、状态管理设计

### 4.1 全局状态（Zustand）

```typescript
// stores/trip-store.ts

import { create } from 'zustand';

interface TripStore {
  // 当前正在创建的行程配置
  currentConfig: Partial<TripConfig>;
  setConfig: (config: Partial<TripConfig>) => void;
  resetConfig: () => void;

  // 生成的行程数据
  currentTrip: Trip | null;
  setCurrentTrip: (trip: Trip | null) => void;

  // 生成进度
  generationProgress: number;
  setGenerationProgress: (progress: number) => void;

  // 城市知识库缓存
  cityKnowledgeCache: Record<string, CityKnowledge>;
  setCityKnowledge: (cityName: string, data: CityKnowledge) => void;
  getCityKnowledge: (cityName: string) => CityKnowledge | null;
}

export const useTripStore = create<TripStore>((set, get) => ({
  currentConfig: {},
  setConfig: (config) =>
    set((state) => ({
      currentConfig: { ...state.currentConfig, ...config },
    })),
  resetConfig: () => set({ currentConfig: {} }),

  currentTrip: null,
  setCurrentTrip: (trip) => set({ currentTrip: trip }),

  generationProgress: 0,
  setGenerationProgress: (progress) => set({ generationProgress: progress }),

  cityKnowledgeCache: {},
  setCityKnowledge: (cityName, data) =>
    set((state) => ({
      cityKnowledgeCache: { ...state.cityKnowledgeCache, [cityName]: data },
    })),
  getCityKnowledge: (cityName) => {
    const state = get();
    return state.cityKnowledgeCache[cityName] || null;
  },
}));
```

### 4.2 服务器状态（TanStack Query）

```typescript
// hooks/use-trip.ts

import { useQuery, useMutation } from '@tanstack/react-query';
import { api } from '@/lib/api-client';

// 查询行程列表
export function useTripList(userId?: number) {
  return useQuery({
    queryKey: ['trips', userId],
    queryFn: () => api.getTripList({ userId }),
    enabled: !!userId,
  });
}

// 查询行程详情
export function useTripDetail(tripId: number) {
  return useQuery({
    queryKey: ['trip', tripId],
    queryFn: () => api.getTrip(tripId),
  });
}

// 生成行程（Mutation）
export function useGenerateTrip() {
  return useMutation({
    mutationFn: (config: TripConfig) => api.generateTrip(config),
    onSuccess: (data) => {
      // 生成成功后，跳转到行程详情页
      window.location.href = `/trip/${data.tripId}`;
    },
  });
}

// 查询交通信息
export function useTransportSearch(params: TransportSearchRequest) {
  return useQuery({
    queryKey: ['transport', params],
    queryFn: () => api.searchTransport(params),
    enabled: !!params.fromCity && !!params.toCity,
    staleTime: 5 * 60 * 1000, // 5 分钟内不重新请求
  });
}
```

## 五、组件交互流程

### 5.1 创建行程流程

```
用户访问 /trip/new
  │
  ▼
显示行程配置页（TripConfigPage）
  │
  ▼
用户填写配置（城市、日期、交通方式、预算...）
  │ （状态存储在 useTripStore.currentConfig）
  ▼
用户点击"生成攻略"
  │
  ▼
前端验证配置（必填项检查）
  │
  ├─ 验证失败 → 显示错误提示
  │
  ▼
验证通过 → 跳转到 /loading
  │
  ▼
显示进度展示页（LoadingPage）
  │
  ▼
前端调用 useGenerateTrip Mutation
  │
  ▼
后端开始生成（SSE 流式响应）
  │
  ▼
前端实时更新进度（useTripStore.generationProgress）
  │
  ▼
生成完成 → 跳转到 /trip/[tripId]
  │
  ▼
显示行程详情页（TripDetailPage）
```
