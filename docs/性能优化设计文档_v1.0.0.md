## 1. 概述

### 1.1 文档目的

本文档定义 PATH-WISE 旅游攻略生成平台的性能优化策略，包括：

- 前端性能优化（加载速度、渲染性能）
- 后端性能优化（API 响应时间、并发处理）
- 数据库性能优化（查询优化、索引策略）
- 缓存策略（Redis 缓存设计）
- CDN 配置（静态资源加速）

### 1.2 性能优化原则

| 原则             | 说明                                   |
| ---------------- | -------------------------------------- |
| **测量优先**     | 先测量性能瓶颈，再优化（避免过早优化） |
| **关键路径优先** | 优先优化用户关键路径（如攻略生成）     |
| **缓存为王**     | 合理使用缓存，减少重复计算             |
| **渐进增强**     | 先保证功能正确，再优化性能             |
| **监控驱动**     | 建立性能监控，持续优化                 |

```typescript
import { useDebouncedCallback } from 'use-debounce';

function SearchInput({ onSearch }: { onSearch: (query: string) => void }) {
  const [query, setQuery] = useState('');

  // 防抖 300ms
  const debouncedSearch = useDebouncedCallback((value: string) => {

// 滚动事件：节流（避免滚动时频繁触发）
import { useThrottledCallback } from 'use-debounce';

function InfiniteScrollList({ onLoadMore }: { onLoadMore: () => void }) {
  const handleScroll = useThrottledCallback(() => {
    const scrollHeight = document.documentElement.scrollHeight;
    const scrollTop = document.documentElement.scrollTop;
    const clientHeight = document.documentElement.clientHeight;

    // 滚动到距离底部 200px 时加载更多
    if (scrollHeight - scrollTop - clientHeight < 200) {
      onLoadMore();
    }
  }, 200);  // 节流 200ms

  useEffect(() => {
    window.addEventListener('scroll', handleScroll);
    return () => window.removeEventListener('scroll', handleScroll);
  }, [handleScroll]);

  return <div>...</div>;
}
```

```typescript
async function getPlansWithDaysBad(userId: string) {

    where: { userId },

async function getPlansWithDaysGood(userId: string) {
  const plans = await prisma.plan.findMany({
    where: { userId },
    include: {
      days: {
        orderBy: { dayIndex: 'asc' },

    },
  });

  return plans;
}

// ✅ 更优：使用 select（只查询需要的字段）
async function getPlansWithDaysOptimized(userId: string) {

      id: true,
      title: true,
      startDate: true,

        select: {
          id: true,
          dayIndex: true,
          theme: true,
        },
        orderBy: { dayIndex: 'asc' },
```

```typescript
// ✅ 推荐：基于游标的分页（性能稳定）
async function getAttractionsCursorBased(
  cityId: string,
  first: number = 20,
  after?: string, // 上一页最后一条记录的 ID
) {
  const attractions = await prisma.attraction.findMany({
    where: {
      cityId,
      ...(after && { id: { gt: after } }), // 只查询 ID > after 的记录
    },
    take: first,
    orderBy: { id: 'asc' },
  });

  const lastAttraction = attractions[attractions.length - 1];

  return {
    edges: attractions.map((attr) => ({
      node: attr,
      cursor: attr.id,
    })),
    pageInfo: {
      hasNextPage: attractions.length === first,
      endCursor: lastAttraction?.id,
    },
  };
}

// ❌ 不推荐：基于 OFFSET 的分页（数据量大时性能差）
async function getAttractionsOffsetBased(cityId: string, page: number = 1, pageSize: number = 20) {
  const skip = (page - 1) * pageSize;

  // OFFSET 在大数据量时性能差（需要扫描并跳过前 N 条记录）
  const attractions = await prisma.attraction.findMany({
    where: { cityId },
    skip,
    take: pageSize,
    orderBy: { id: 'asc' },
  });

  return attractions;
}
```

## 6. 缓存策略

### 6.1 缓存层级

```
请求 → CDN 缓存 → Nginx 缓存 → 应用缓存（Redis）→ 数据库缓存
```

### 6.2 Redis 缓存设计

#### 6.2.1 缓存键命名规范

```
# 格式：<业务>:<实体>:<ID>:<字段>（可选）

# 示例
city:attractions:北京:list          # 北京的所有景点列表
city:attractions:北京:detail:001    # 北京景点 ID=001 的详情
plan:user:user_001:list            # 用户 user_001 的所有行程
plan:detail:plan_001               # 行程 plan_001 的详情
config:system:transport_buffer     # 系统配置：交通缓冲时间
```

#### 6.2.2 缓存策略

| 数据类型         | 缓存策略                     | TTL     |
| ---------------- | ---------------------------- | ------- |
| **城市知识库**   | 预加载（启动时加载到 Redis） | 24 小时 |
| **景点详情**     | 懒加载（首次查询时缓存）     | 1 小时  |
| **用户行程**     | 写失效（更新时清除缓存）     | 30 分钟 |
| **API 响应**     | 响应缓存（GET 请求）         | 5 分钟  |
| **LLM 生成结果** | 写缓存（生成成功后缓存）     | 7 天    |

#### 6.2.3 缓存实现示例

```typescript
// apps/api/src/services/cache.service.ts
import Redis from 'ioredis';

const redis = new Redis(process.env.REDIS_URL);

export async function getOrSetCache<T>(
  key: string,
  ttlSeconds: number,
  fetchFn: () => Promise<T>,
): Promise<T> {
  // 1. 尝试从缓存读取
  const cached = await redis.get(key);
  if (cached) {
    return JSON.parse(cached) as T;
  }

  // 2. 缓存未命中，调用 fetchFn 获取数据
  const data = await fetchFn();

  // 3. 写入缓存
  await redis.setex(key, ttlSeconds, JSON.stringify(data));

  return data;
}

// 使用缓存服务
async function getCityAttractions(cityId: string) {
  return getOrSetCache(
    `city:attractions:${cityId}:list`,
    3600, // 缓存 1 小时
    async () => {
      // 从数据库查询
      return await prisma.attraction.findMany({
        where: { cityId },
      });
    },
  );
}
```

### 6.3 缓存穿透/击穿/雪崩防护

```typescript
// 1. 缓存穿透防护（查询不存在的数据）
async function getAttractionDetail(id: string) {
  // 使用布隆过滤器（Bloom Filter）预先判断 ID 是否存在
  if (!(await bloomFilter.exists(id))) {
    throw new Error('Attraction not found');
  }

  return getOrSetCache(`attraction:detail:${id}`, 3600, async () => {
    const attraction = await prisma.attraction.findUnique({
      where: { id },
    });

    if (!attraction) {
      // 缓存空值（避免频繁查询不存在的 ID）
      await redis.setex(`attraction:detail:${id}`, 60, JSON.stringify(null));
      throw new Error('Attraction not found');
    }

    return attraction;
  });
}

// 2. 缓存击穿防护（热点 Key 过期）
import Redlock from 'redlock';

const redlock = new Redlock([redis], {
  driftFactor: 0.01,
  retryCount: 3,
  retryDelay: 200,
});

async function getHotAttractionDetail(id: string) {
  const cacheKey = `attraction:detail:${id}`;

  // 尝试从缓存读取
  const cached = await redis.get(cacheKey);
  if (cached) {
    return JSON.parse(cached);
  }

  // 缓存未命中，获取分布式锁（避免多个请求同时查询数据库）
  const lock = await redlock.acquire([`lock:${cacheKey}`], 1000);

  try {
    // 获取锁成功，查询数据库
    const attraction = await prisma.attraction.findUnique({
      where: { id },
    });

    // 写入缓存
    await redis.setex(cacheKey, 3600, JSON.stringify(attraction));

    return attraction;
  } finally {
    // 释放锁
    await lock.release();
  }
}

// 3. 缓存雪崩防护（大量 Key 同时过期）
async function setCacheWithRandomTTL(key: string, value: any, baseTTL: number) {
  // 随机 TTL（避免大量 Key 同时过期）
  const randomTTL = baseTTL + Math.floor(Math.random() * 300); // 加 0-5 分钟随机
  await redis.setex(key, randomTTL, JSON.stringify(value));
}
```

```typescript
// apps/api/src/plugins/metrics.ts
import fastify from 'fastify';
import promClient from 'prom-client';

const register = new promClient.Registry();

// 注册默认指标（Node.js 运行时指标）
promClient.collectDefaultMetrics({ register });

// 自定义指标：API 请求耗时
const httpRequestDuration = new promClient.Histogram({
  name: 'http_request_duration_seconds',
  help: 'Duration of HTTP requests in seconds',
  labelNames: ['method', 'route', 'status_code'],
  buckets: [0.1, 0.3, 0.5, 0.7, 1, 3, 5, 7, 10],
});

register.registerMetric(httpRequestDuration);

// 中间件：记录请求耗时
export function metricsMiddleware(app: FastifyInstance) {
  app.addHook('onRequest', async (request, reply) => {
    request.startTime = process.hrtime();
  });

  app.addHook('onResponse', async (request, reply) => {
    const duration = process.hrtime(request.startTime);
    const durationSeconds = duration[0] + duration[1] / 1e9;

    httpRequestDuration
      .labels(request.method, request.routeOptions.url, reply.statusCode.toString())
      .observe(durationSeconds);
  });
}

// 暴露 /metrics 端点（Prometheus 抓取）
app.get('/metrics', async (request, reply) => {
  reply.header('Content-Type', register.contentType);
  return register.metrics();
});
```
